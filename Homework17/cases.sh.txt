#/bin/bash

printf 'Введите букву,цифру или символ!'
read DISTR
case $DISTR in
        [[:upper:]])
        echo "Вы ввели букву в вверхнем регистре. О,любите побольше))"
        ;;
        [[:lower:]])
        echo "Эй! Это нижний регистр! Уверены?"
        ;;
        [0-9])
        echo "Совсем просто - Вы ввели всем известную арабскую цифру."
        ;;
        *)
        echo "Хм, редкий выбор."
        ;;
esac
